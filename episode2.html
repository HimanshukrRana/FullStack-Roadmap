4. Hooks & useEffect

useEffect(() => {
  let cancelled = false;
  const controller = new AbortController();
  fetchUser(userId).then(d => !cancelled && setUser(d));
  return () => { cancelled = true; controller.abort(); }; // Cleanup
}, [userId]); // ✅ Dependencies!

Rules of Hooks:
1. Only call at top level
2. Only from React functions
3. Never in loops/conditions/nested functions

5. Component Architecture

Smart vs Dumb:
• Smart: Fetches data, manages state, passes callbacks
• Dumb: Receives props, renders UI, emits events

// Smart Container
function UserDashboard({ userId }) {
  const { data } = useFetch(`/api/users/${userId}`);
  return <DashboardView user={data} onUpdate={handleUpdate} />;
}

// Dumb View
function DashboardView({ user, onUpdate }) {
  return <ProfileForm values={user} onSubmit={onUpdate} />;
}

import { create } from 'zustand';
const useCart = create(set => ({
  items: [],
  addItem: p => set(s => ({ items: [...s.items, p] })),
  total: (s) => s.items.reduce((a,b) => a+b.price, 0)

8. Next.js Mastery

graph LR A[Request] --> B{Static?} -->|Yes| C[SSG/ISR] B -->|No| D[SSR] D --> E[Client Component] C & D --> F[HTML + JS Bundle]
// Server Component (default)
export default async function Page({ params }) {
  const data = await db.post.find(params.slug); // Direct DB!
  return <Post post={data} />;
}

// Client Component
'use client';
export function Counter() {
  const [count, setCount] = useState(0);
  return <button onClick={() => setCount(c => c+1)}>{count}</button>
}

Rendering Strategies:
• SSG: Pre-build (fastest)
• SSR: Build per request (fresh)
• ISR: Update static pages without full rebuild

9. Advanced Forms

// React Hook Form + Zod
import { useForm } from 'react-hook-form';
import { zodResolver } from '@hookform/resolvers/zod';
import { userSchema } from '@/schemas';

const { register, handleSubmit, formState: { errors } } = useForm({
  resolver: zodResolver(userSchema)
});

<form onSubmit={handleSubmit(onSubmit)}>
  <input {...register('email')} />
  {errors.email && <p>{errors.email.message}</p>}
  <button type="submit">Save</button>
</form>
