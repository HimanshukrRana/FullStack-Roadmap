2. Docker & Compose

# Dockerfile (Multi-stage)
FROM node:18-alpine AS deps
WORKDIR /app
COPY package*.json ./
RUN npm ci --only=production

FROM node:18-alpine AS production
COPY --from=deps /app/node_modules ./node_modules
COPY --from=builder /app/dist ./dist
USER nodejs
EXPOSE 80
CMD ["node", "dist/server.js"]
# docker-compose.yml
services:
  app:
    build: .
    ports: ["3000:80"]
    depends_on: [mongo, redis]
  mongo:
    image: mongo:6
    volumes: [mongo-data:/data/db]
  redis:
    image: redis:7-alpine

graph LR A[Internet] -->|443| B[ALB] -->|80| C[EC2 Private] -->|27017| D[RDS Private]
# Bootstrap (user-data.sh)
yum update -y
curl -fsSL https://rpm.nodesource.com/setup_18.x | bash -

🔐 Security Groups: Least privilege. ALB → 443. EC2 → 80 from ALB only. DB → 27017 from EC2 only. SSH → Bastion IP only.

4. Load Balancing & Auto Scaling

# Terraform: ALB + ASG
resource "aws_autoscaling_group" "app" {
  min_size = 2; max_size = 10
  target_group_arns = [aws_lb_target_group.app.arn]
  health_check_type = "ELB"
}

resource "aws_autoscaling_policy" "cpu" {
  target_tracking_configuration {
    predefined_metric_specification { predefined_metric_type = "ASGAverageCPUUtilization" }
    target_value = 70.0
  }
}

const cacheControl = {
  'js':'public, max-age=31536000, immutable',
  'css':'public, max-age=31536000, immutable',
  'html':'public, max-age=300, stale-while-revalidate=3600'
};
res.set('Cache-Control', cacheControl[ext] || 'no-store');
🌍 Cache Invalidation: Versioned filenames (best), query busting, or CloudFront API (limited free tier).

6. CI/CD with GitHub Actions

# .github/workflows/deploy.yml
name: Deploy
on: [push.main]
jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - run: npm ci && npm test
  deploy:
    needs: test
    runs-on: ubuntu-latest
    steps:
      - uses: aws-actions/configure-aws-credentials@v4
      - run: aws ecs update-service --cluster prod --service api --force-new-deployment

2. Check CloudWatch Dashboard

5. Post-mortem: Root cause + prevention
